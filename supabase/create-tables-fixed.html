Step 1: Create Tables in Supabase

Follow these steps to create the necessary tables:

  1. Go to your Supabase Dashboard
  2. Select your project
  3. Click on the SQL Editor in the left sidebar
  4. Click New Query
  5. Copy and paste the SQL code from the right panel
  6. Click Run to execute the query

Required Tables for PlanetSham

  • users - Store user information
  • posts - Store posts content
  • trending_topics - Store trending data
  • follows - Manage user relationships

SQL Code for Tables

CREATE TABLE users ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, username TEXT UNIQUE NOT NULL, full_name TEXT, avatar_url TEXT, created_at TIMESTAMP WITH TIME ZONE DEFAULT now() ); CREATE TABLE posts ( id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, user_id UUID REFERENCES users(id), content TEXT NOT NULL, likes BIGINT DEFAULT 0, created_at TIMESTAMP WITH TIME ZONE DEFAULT now() ); CREATE TABLE trending_topics ( id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, name TEXT NOT NULL, post_count BIGINT DEFAULT 0, rank INTEGER, created_at TIMESTAMP WITH TIME ZONE DEFAULT now() ); CREATE TABLE follows ( id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, follower_id UUID REFERENCES users(id), following_id UUID REFERENCES users(id), created_at TIMESTAMP WITH TIME ZONE DEFAULT now(), UNIQUE(follower_id, following_id) );

Enable Row Level Security

-- Enable Row Level Security on all tables ALTER TABLE users ENABLE ROW LEVEL SECURITY; ALTER TABLE posts ENABLE ROW LEVEL SECURITY; ALTER TABLE trending_topics ENABLE ROW LEVEL SECURITY; ALTER TABLE follows ENABLE ROW LEVEL SECURITY; -- Create policies to allow read access to all CREATE POLICY "Allow public read access" ON users FOR SELECT USING (true); CREATE POLICY "Allow public read access" ON posts FOR SELECT USING (true); CREATE POLICY "Allow public read access" ON trending_topics FOR SELECT USING (true); CREATE POLICY "Allow public read access" ON follows FOR SELECT USING (true);
